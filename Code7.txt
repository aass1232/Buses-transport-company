CREATE TABLE [Drivers] (
  [DriverID] INT NOT NULL,
  [FirstName] VARCHAR(50),
  [LastNam] VARCHAR(50),
  [Salary] INT NOT NULL,
  [LineID] INT NOT NULL,
  PRIMARY KEY ([DriverID])
);

CREATE TABLE [Lines] (
  [LineID] INT NOT NULL,
  [Direction] VARCHAR(50),
  [Stops] VARCHAR(50),
  [Start] VARCHAR(50),
  [End] VARCHAR(50),
  PRIMARY KEY ([LineID])
);

CREATE TABLE [Tiket] (
  [TiketID] INT NOT NULL,
  [Price] INT NOT NULL,
  [Line] VARCHAR(50),
  PRIMARY KEY ([TiketID])
);

CREATE INDEX [FK] ON  [Tiket] ([Line]);

CREATE TABLE [Bus] (
  [BusID] ITN NOT NULL,
  [Line] INT NOT NULL,
  [DriverID] VARCHAR(50),
  [Collector] VARCHAR(50),
  [TiketID] INT NOT NULL,
  PRIMARY KEY ([BusID])
);

CREATE INDEX [FK] ON  [Bus] ([Line], [DriverID], [TiketID]);

CREATE TABLE [Managers] (
  [ManagerID] INT NOT NULL,
  [Name] VARCHAR(50),
  [Phone] ITN NOT NULL,
  PRIMARY KEY ([ManagerID])
);
